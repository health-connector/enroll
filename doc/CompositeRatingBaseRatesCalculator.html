def tier_rates
  base_rate_value = base_rate
  tier_rate_values = {}
  CompositeRatingTier::NAMES.each do |crt|
    crt_rate = benefit_group.composite_rating_tier_factor_for(crt, plan) * base_rate_value
    tier_rate_values[crt] = crt_rate.round(2)
  end
  tier_rate_values
end
