def self.run
  conn = Bunny.new(Rails.application.config.acapi.remote_broker_uri, :heartbeat => 15)
  conn.start
  ch = conn.create_channel
  ch.prefetch(1)
  q = ch.queue(queue_name, :durable => true)

  self.new(ch, q).subscribe(:block => true, :manual_ack => true)
  conn.close
end
