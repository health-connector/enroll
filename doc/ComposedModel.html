def validate_collection_and_propagate_errors(collection_name, objs)
  objs.each_with_index do |obj, idx|
    obj.valid?
    obj.errors.each do |attr, err|
      errors.add(validate_collection_error_key_name(collection_name, idx, attr), err)
    end
  end
end
